function update(){ if(!gameActive) return; // תנועה player.dx = 0; if(keys["ArrowRight"] || keys["KeyD"]) player.dx = player.speed; if(keys["ArrowLeft"] || keys["KeyA"]) player.dx = -player.speed; if((keys["Space"] || keys["ArrowUp"] || keys["KeyW"]) && player.onGround){ player.dy = player.jump; player.onGround = false; } player.dy += player.gravity; player.x += player.dx; player.y += player.dy; player.onGround = false; // שובל תנועה player.trail.push({x: player.x, y: player.y, alpha: 0.5}); if(player.trail.length > 8) player.trail.shift(); // התנגשות במשטחים platforms.forEach(p => { if(player.x < p.x + p.w && player.x + player.w > p.x && player.y + player.h < p.y + 15 && player.y + player.h + player.dy >= p.y){ player.y = p.y - player.h; player.dy = 0; player.onGround = true; } }); // --- תיקון הפסילה --- enemies.forEach(e => { e.x += e.dir * 2.5; if(Math.abs(e.x - e.startX) > e.range) e.dir *= -1; if(player.x < e.x + e.w && player.x + player.w > e.x && player.y < e.y + e.h && player.y + player.h > e.y){ // ריסט מלא - גם לשחקן וגם למצלמה player.x = 100; player.y = 300; player.dy = 0; scrollX = 0; // חזרה של המצלמה לתחילת השלב // אפקט ויזואלי של פסילה (אופציונלי) console.log("סוכן נוטרל! חוזר לנקודת התחלה..."); } }); // איסוף מטבעות coins.forEach(c => { if(!c.collected && Math.hypot(player.x - c.x, player.y - c.y) < 30){ c.collected = true; } }); // עדכון מצלמה - עוקבת אחרי השחקן רק אם הוא עבר את חצי המסך if(player.x > 300) { scrollX = player.x - 300; } else { scrollX = 0; // אם השחקן חזר להתחלה, גם המצלמה מתאפסת } if(player.x > levelEnd) handleWin(); // בדיקה אם השחקן נפל מהמסך (נפילה לתהום) if(player.y > canvas.height) { player.x = 100; player.y = 300; player.dy = 0; scrollX = 0; } }
